# VM调度系统工业级编程手册

## 手册概述

### 1.1 编写目的

本手册提供VM（虚拟机）调度系统的工业级实现规范，聚焦**实时性、资源隔离、容错性、公平性**核心目标，摒弃练手级简化设计，覆盖CPU调度、中断调度、外设调度全场景的最优实现方案。

### 1.2 适用范围

本手册适用于基于C++开发的Linux环境下VM调度系统，核心模块包括`BaseVM`（VM基类）、`mVM`（VM实例类）、`VmManager`（调度核心管理器）。

### 1.3 核心目标

|目标维度|具体要求|
|---|---|
|实时性|同步中断响应延迟≤10ms（硬实时）|
|资源隔离|单VM CPU/外设占用率可精准管控|
|容错性|故障VM快速隔离，不扩散影响|
|公平性|多VM资源占用无抢占、无饥饿|
## 第二章 核心设计：VM状态与调度上下文

### 2.1 VM状态模型（工业级）

状态枚举定义在`baseVM.h`，是调度算法的核心决策依据，禁止随意扩展/修改状态值。

```C++

// baseVM.h - 不可修改状态值，扩展需评审
enum class VMState {
    INIT = 0,       // 初始化中（不可调度，资源未就绪）
    READY = 1,      // 就绪（等待CPU/外设，可进入调度队列）
    RUNNING = 2,    // 运行中（占用资源，实时更新使用量）
    BLOCKED = 3,    // 阻塞（等待外设/中断，暂不可调度）
    SUSPENDED = 4,  // 挂起（同步中断触发，高优调度）
    FAULT = 5,      // 故障（资源超限/超时，强制隔离）
    TERMINATED = 6  // 终止（资源已释放，清理调度队列）
};
```

#### 2.1.1 状态转换规则（强制约束）

|源状态|目标状态|触发条件|
|---|---|---|
|INIT|READY|VM初始化完成，资源绑定成功|
|READY|RUNNING|调度器选中，分配CPU/外设资源|
|RUNNING|BLOCKED|发起外设请求/等待中断，主动释放资源|
|BLOCKED|READY|外设响应/中断完成，资源重新可用|
|RUNNING|SUSPENDED|同步高优中断触发，强制挂起|
|SUSPENDED|RUNNING|调度器高优选中，抢占CPU资源|
|任意状态|FAULT|资源使用超限/超时/异常，容错机制触发|
|FAULT/READY|TERMINATED|人工终止/自动清理，资源全量释放|
### 2.2 调度上下文结构

在`baseVM.h`中定义调度算法所需的核心上下文，所有字段需实时更新，禁止只读不写。

```C++

// baseVM.h - 调度上下文（算法决策核心数据）
struct VMScheduleCtx {
    uint64_t deadline;          // EDF调度：截止时间（ns），同步中断=当前+10ms
    uint32_t priority;          // 静态优先级（兜底，1-10，1最高）
    uint64_t cpu_budget;        // CFS带宽：单周期CPU最大使用量（us，如200000=200ms/1s）
    uint64_t cpu_used;          // 已用CPU时间（us，实时累加）
    int interrupt_level;        // 中断等级（1=同步高优，2=异步低优）
    int blocked_reason;         // 阻塞原因（0=无，1=外设，2=中断，3=资源不足）
    uint64_t timeout_ts;        // 超时时间戳（ns，超时触发FAULT）
    double peripheral_share;    // Fair Share：外设使用权份额（0.0-1.0）
    uint64_t peripheral_used;   // 已用外设时间（us）
};

// BaseVM类扩展（状态+上下文访问接口）
class BaseVM {
private:
    VMState state;
    VMScheduleCtx sched_ctx;
    pid_t vm_pid;               // VM进程ID（绑定Linux调度器）
    cpu_set_t cpu_affinity;     // CPU亲和性集合
public:
    // 状态操作（禁止直接赋值，仅通过接口）
    inline VMState getState() const { return state; }
    void transitionTo(VMState target_state, int blocked_reason = 0);
    
    // 上下文访问（读写分离）
    inline const VMScheduleCtx& getSchedCtx() const { return sched_ctx; }
    inline VMScheduleCtx& mutableSchedCtx() { return sched_ctx; }
    
    // 资源绑定接口
    inline pid_t getPid() const { return vm_pid; }
    void setCPUAffinity(const cpu_set_t& cpuset);
};
```

## 第三章 调度算法实现（核心章节）

### 3.1 CPU调度（最优：CFS+CPU亲和性+带宽控制）

#### 3.1.1 算法原理

- CFS（完全公平调度）：通过`vruntime`（虚拟运行时间）保证多VM CPU占用绝对公平，避免饥饿；

- CPU亲和性：绑定VM到固定CPU核，减少跨核调度开销，提升缓存命中率；

- CFS Bandwidth Control：限制单VM CPU使用率，故障VM无法耗尽资源。

#### 3.1.2 代码实现（VmManager.cpp）

```C++

// VmManager.h 前置声明
#include <sched.h>
#include <vector>
#include <memory>
#include "baseVM.h"

class VmManager {
private:
    std::vector<std::shared_ptr<BaseVM>> vm_list; // 管理所有VM实例
    cpu_set_t default_cpuset;                     // 默认CPU亲和性集合
    struct sched_attr cfs_attr;                   // CFS调度属性
    
    // 内部方法：更新VM的CFS带宽
    void updateCFSBandwidth(const std::shared_ptr<BaseVM>& vm);
public:
    // 初始化CFS调度器（进程启动时调用）
    void initCFSScheduler(int cpu_core_num);
    // 选择下一个待调度的VM（CFS核心逻辑）
    std::shared_ptr<BaseVM> selectNextVM();
    // 实时更新VM的CPU使用量
    void updateVMCPUUsage(const std::shared_ptr<BaseVM>& vm);
};

// VmManager.cpp - CFS调度器初始化
void VmManager::initCFSScheduler(int cpu_core_num) {
    // 初始化CPU亲和性：按核数分配（如4核则VM0→0核，VM1→1核...）
    CPU_ZERO(&default_cpuset);
    for (int i = 0; i < cpu_core_num; i++) {
        CPU_SET(i, &default_cpuset);
    }

    // 初始化CFS属性：带宽控制（周期1s，单VM最大200ms）
    cfs_attr.size = sizeof(struct sched_attr);
    cfs_attr.sched_policy = SCHED_CFS;
    cfs_attr.sched_nice = 0;
    cfs_attr.sched_runtime = 200000000; // 200ms/周期（ns）
    cfs_attr.sched_period = 1000000000; // 周期1s（ns）
    cfs_attr.sched_deadline = 0;

    // 为所有VM绑定CFS属性和CPU亲和性
    for (size_t i = 0; i < vm_list.size(); i++) {
        auto vm = vm_list[i];
        // 绑定CPU核（轮询分配）
        cpu_set_t cpuset;
        CPU_ZERO(&cpuset);
        CPU_SET(i % cpu_core_num, &cpuset);
        vm->setCPUAffinity(cpuset);
        sched_setaffinity(vm->getPid(), sizeof(cpu_set_t), &cpuset);

        // 绑定CFS带宽
        sched_setattr(vm->getPid(), &cfs_attr, 0);
        // 初始化CPU预算
        vm->mutableSchedCtx().cpu_budget = 200000; // 200ms/1s（us）
    }
}

// VmManager.cpp - CFS核心：选择下一个VM（按vruntime排序）
std::shared_ptr<BaseVM> VmManager::selectNextVM() {
    auto it = std::min_element(vm_list.begin(), vm_list.end(),
        [](const std::shared_ptr<BaseVM>& a, const std::shared_ptr<BaseVM>& b) {
            // 过滤不可调度状态
            if (a->getState() == VMState::FAULT || a->getState() == VMState::TERMINATED || a->getState() == VMState::INIT)
                return false;
            if (b->getState() == VMState::FAULT || b->getState() == VMState::TERMINATED || b->getState() == VMState::INIT)
                return true;

            // SUSPENDED态（同步中断）优先调度
            if (a->getState() == VMState::SUSPENDED && b->getState() != VMState::SUSPENDED)
                return true;
            if (b->getState() == VMState::SUSPENDED && a->getState() != VMState::SUSPENDED)
                return false;

            // 核心：按cpu_used（vruntime）排序，越小越优先
            return a->getSchedCtx().cpu_used < b->getSchedCtx().cpu_used;
        });
    return (it != vm_list.end()) ? *it : nullptr;
}

// VmManager.cpp - 实时更新VM CPU使用量（定时器触发，10ms/次）
void VmManager::updateVMCPUUsage(const std::shared_ptr<BaseVM>& vm) {
    if (vm->getState() != VMState::RUNNING) return;
    // 读取/proc/[pid]/stat获取实际CPU使用量（工业级精准方式）
    char stat_path[256];
    snprintf(stat_path, sizeof(stat_path), "/proc/%d/stat", vm->getPid());
    FILE* fp = fopen(stat_path, "r");
    if (!fp) return;

    uint64_t utime, stime;
    fscanf(fp, "%*d %*s %*c %*d %*d %*d %*d %*d %*u %*u %*u %*u %*u %lu %lu", &utime, &stime);
    fclose(fp);

    // 转换为us（utime/stime单位是jiffies，需乘以jiffy时长）
    uint64_t cpu_used = (utime + stime) * (1000000 / sysconf(_SC_CLK_TCK));
    vm->mutableSchedCtx().cpu_used = cpu_used;

    // 超限检查：触发FAULT态
    if (cpu_used > vm->getSchedCtx().cpu_budget) {
        vm->transitionTo(VMState::FAULT, 3); // 3=资源不足
    }
}
```

### 3.2 中断调度（最优：EDF+优先级继承）

#### 3.2.1 算法原理

- EDF（最早截止时间优先）：保证同步中断的硬实时性，始终调度`deadline`最早的VM；

- 优先级继承：解决优先级反转问题，低优VM占用高优VM资源时，临时提升低优VM优先级。

#### 3.2.2 代码实现（VmManager.cpp）

```C++

// VmManager.cpp - 处理中断请求（中断模块回调）
void VmManager::handleInterrupt(std::shared_ptr<BaseVM> vm, int interrupt_level) {
    auto& ctx = vm->mutableSchedCtx();
    ctx.interrupt_level = interrupt_level;
    uint64_t now = std::chrono::duration_cast<std::chrono::nanoseconds>(
        std::chrono::system_clock::now().time_since_epoch()).count();

    // 设置EDF截止时间：同步高优（10ms），异步低优（100ms）
    if (interrupt_level == 1) { // 同步高优
        ctx.deadline = now + 10 * 1000 * 1000; // 10ms（ns）
        ctx.timeout_ts = now + 15 * 1000 * 1000; // 超时15ms触发FAULT
        vm->transitionTo(VMState::SUSPENDED);
    } else { // 异步低优
        ctx.deadline = now + 100 * 1000 * 1000; // 100ms（ns）
        ctx.timeout_ts = now + 120 * 1000 * 1000; // 超时120ms触发FAULT
        vm->transitionTo(VMState::READY);
    }

    // 优先级继承检查：解决优先级反转
    checkPriorityInheritance(vm);
}

// VmManager.cpp - 优先级继承检查
void VmManager::checkPriorityInheritance(const std::shared_ptr<BaseVM>& high_prio_vm) {
    // 遍历所有VM，查找占用高优VM所需资源的低优VM
    for (auto& vm : vm_list) {
        if (vm->getState() != VMState::RUNNING) continue;
        // 检查资源占用（示例：外设GIL、共享内存等）
        if (isOccupyingResource(vm, high_prio_vm)) {
            // 临时提升优先级至高优VM的优先级
            vm->mutableSchedCtx().priority = high_prio_vm->getSchedCtx().priority;
            // 更新Linux进程优先级
            setpriority(PRIO_PROCESS, vm->getPid(), -vm->getSchedCtx().priority);
        }
    }
}

// VmManager.cpp - EDF调度选择（中断场景专用）
std::shared_ptr<BaseVM> VmManager::selectEDFVM() {
    auto it = std::min_element(vm_list.begin(), vm_list.end(),
        [](const std::shared_ptr<BaseVM>& a, const std::shared_ptr<BaseVM>& b) {
            // 过滤不可调度状态
            if (a->getState() == VMState::FAULT || a->getState() == VMState::TERMINATED)
                return false;
            if (b->getState() == VMState::FAULT || b->getState() == VMState::TERMINATED)
                return true;

            // 超时检查：触发FAULT
            uint64_t now = std::chrono::duration_cast<std::chrono::nanoseconds>(
                std::chrono::system_clock::now().time_since_epoch()).count();
            if (a->getSchedCtx().timeout_ts > 0 && a->getSchedCtx().timeout_ts < now) {
                a->transitionTo(VMState::FAULT, 2); // 2=中断超时
                return false;
            }
            if (b->getSchedCtx().timeout_ts > 0 && b->getSchedCtx().timeout_ts < now) {
                b->transitionTo(VMState::FAULT, 2);
                return true;
            }

            // EDF核心：截止时间越早越优先
            return a->getSchedCtx().deadline < b->getSchedCtx().deadline;
        });
    return (it != vm_list.end()) ? *it : nullptr;
}
```

### 3.3 外设调度（最优：MLFQ+Fair Share+GIL精细化管理）

#### 3.3.1 算法原理

- MLFQ（多级反馈队列）：按外设请求优先级分层调度，高优队列短时间片、抢占式；

- Fair Share：按份额分配外设使用权，超份额则降级，保证公平；

- GIL精细化管理：按请求粒度释放GIL，高优请求可抢占，避免长期占用。

#### 3.3.2 代码实现（VmManager.cpp）

```C++

// VmManager.h 扩展外设调度相关成员
class VmManager {
private:
    std::queue<std::shared_ptr<BaseVM>> mlfq_queues[3]; // MLFQ队列：0=高优,1=中优,2=低优
    std::shared_ptr<BaseVM> gil_holder;                 // 当前GIL持有者
    uint64_t gil_acquire_ts;                            // GIL获取时间戳（ns）
    const uint64_t gil_timeout = 10 * 1000 * 1000;      // GIL超时10ms（ns）

    // 内部方法：检查Fair Share份额
    bool isOverPeripheralShare(const std::shared_ptr<BaseVM>& vm);
    // 释放GIL（主动/强制）
    void releaseGIL(const std::shared_ptr<BaseVM>& vm, bool force);
public:
    // 外设请求入队
    void enqueuePeripheralReq(std::shared_ptr<BaseVM> vm);
    // 选择下一个外设调度的VM
    std::shared_ptr<BaseVM> selectPeripheralVM();
    // 获取/释放GIL
    bool acquireGIL(const std::shared_ptr<BaseVM>& vm);
};

// VmManager.cpp - 外设请求入队（MLFQ）
void VmManager::enqueuePeripheralReq(std::shared_ptr<BaseVM> vm) {
    int queue_idx = 2; // 默认低优队列
    auto& ctx = vm->getSchedCtx();

    // 按中断等级分队列
    if (ctx.interrupt_level == 1) { // 同步高优→0号队列
        queue_idx = 0;
    } else if (ctx.interrupt_level == 2) { // 异步中优→1号队列
        queue_idx = 1;
    }

    // Fair Share检查：超份额则降级
    if (isOverPeripheralShare(vm)) {
        queue_idx = std::min(queue_idx + 1, 2); // 最多降级到低优
    }

    mlfq_queues[queue_idx].push(vm);
}

// VmManager.cpp - Fair Share份额检查
bool VmManager::isOverPeripheralShare(const std::shared_ptr<BaseVM>& vm) {
    auto& ctx = vm->getSchedCtx();
    // 外设总使用时间（周期1s）
    static uint64_t peripheral_total_used = 0;
    // 份额=peripheral_share（如0.1=10%），超则返回true
    return (ctx.peripheral_used / (double)peripheral_total_used) > ctx.peripheral_share;
}

// VmManager.cpp - 外设调度选择（MLFQ核心）
std::shared_ptr<BaseVM> VmManager::selectPeripheralVM() {
    // 优先调度高优队列，空则下沉
    for (int i = 0; i < 3; ++i) {
        if (mlfq_queues[i].empty()) continue;

        auto vm = mlfq_queues[i].front();
        mlfq_queues[i].pop();

        // GIL抢占逻辑：高优队列可抢占低优GIL持有者
        if (i == 0 && gil_holder != nullptr) {
            if (gil_holder->getSchedCtx().interrupt_level != 1) {
                releaseGIL(gil_holder, true); // 强制释放
            }
        }

        // 尝试获取GIL
        if (acquireGIL(vm)) {
            return vm;
        } else {
            // 获取失败则重新入队（降级一级）
            int requeue_idx = std::min(i + 1, 2);
            mlfq_queues[requeue_idx].push(vm);
        }
    }
    return nullptr;
}

// VmManager.cpp - GIL获取/释放
bool VmManager::acquireGIL(const std::shared_ptr<BaseVM>& vm) {
    uint64_t now = std::chrono::duration_cast<std::chrono::nanoseconds>(
        std::chrono::system_clock::now().time_since_epoch()).count();

    // GIL未被占用，直接获取
    if (gil_holder == nullptr) {
        gil_holder = vm;
        gil_acquire_ts = now;
        return true;
    }

    // GIL已超时，强制释放并获取
    if (now - gil_acquire_ts > gil_timeout) {
        releaseGIL(gil_holder, true);
        gil_holder = vm;
        gil_acquire_ts = now;
        return true;
    }

    // 否则获取失败
    return false;
}

void VmManager::releaseGIL(const std::shared_ptr<BaseVM>& vm, bool force) {
    if (gil_holder != vm && !force) return;

    // 更新外设使用时间
    uint64_t now = std::chrono::duration_cast<std::chrono::nanoseconds>(
        std::chrono::system_clock::now().time_since_epoch()).count();
    vm->mutableSchedCtx().peripheral_used += (now - gil_acquire_ts) / 1000; // 转换为us

    // 释放GIL
    gil_holder = nullptr;
    gil_acquire_ts = 0;
}
```

### 3.4 容错调度（最优：故障隔离+资源回收）

#### 3.4.1 算法原理

- 实时故障检测：定时扫描VM状态，发现`FAULT`态立即隔离；

- 抢占式资源回收：强制剥夺故障VM的CPU/外设/GIL资源；

- 负载补偿：故障VM的资源份额自动分配给正常VM，保证整体吞吐量。

#### 3.4.2 代码实现（VmManager.cpp）

```C++

// VmManager.cpp - 容错调度（定时器触发，100ms/次）
void VmManager::faultToleranceScheduler() {
    std::vector<std::shared_ptr<BaseVM>> fault_vms;
    std::vector<std::shared_ptr<BaseVM>> normal_vms;

    // 分类VM：故障/正常
    for (auto& vm : vm_list) {
        if (vm->getState() == VMState::FAULT) {
            fault_vms.push_back(vm);
        } else if (vm->getState() != VMState::TERMINATED && vm->getState() != VMState::INIT) {
            normal_vms.push_back(vm);
        }
    }

    // 1. 故障VM资源回收
    for (auto& vm : fault_vms) {
        // 强制释放CPU/外设/GIL
        sched_setattr(vm->getPid(), &cfs_attr, 0); // 重置CFS属性
        releaseGIL(vm, true);
        // 标记为隔离态，禁止入队
        vm->transitionTo(VMState::TERMINATED);
    }

    // 2. 负载补偿：分配故障VM的CPU/外设份额
    if (fault_vms.empty() || normal_vms.empty()) return;
    double share_per_vm = 0.0;
    for (auto& fault_vm : fault_vms) {
        share_per_vm += fault_vm->getSchedCtx().peripheral_share;
    }
    share_per_vm /= normal_vms.size();

    for (auto& vm : normal_vms) {
        vm->mutableSchedCtx().peripheral_share += share_per_vm;
        // 提升CPU预算（最多不超过单核80%）
        uint64_t new_budget = std::min(vm->getSchedCtx().cpu_budget + 50000, 800000ULL);
        vm->mutableSchedCtx().cpu_budget = new_budget;
        updateCFSBandwidth(vm); // 更新CFS带宽
    }
}
```

## 第四章 代码集成约束

### 4.1 模块职责划分（强制）

|模块|职责边界|禁止操作|
|---|---|---|
|BaseVM|仅维护VM状态、调度上下文、资源绑定接口|实现调度逻辑、直接修改Linux调度器属性|
|mVM|继承BaseVM，实现VM业务逻辑|重写状态转换接口、修改调度上下文核心字段|
|VmManager|唯一调度入口，实现所有调度算法|直接操作VM业务逻辑、忽略状态转换规则|
### 4.2 接口规范

- 状态转换仅允许通过`transitionTo`接口，禁止直接修改`state`字段；

- 调度上下文修改仅允许通过`mutableSchedCtx()`接口，只读场景用`getSchedCtx()`；

- 所有调度算法的选择逻辑，必须通过`VmManager`的`selectXXXVM`接口暴露，禁止外部直接操作队列。

### 4.3 性能约束

|指标|阈值|检测方式|
|---|---|---|
|调度延迟|≤1ms（CPU调度）|打点`selectNextVM`调用→返回的耗时|

### 6.3 简化调度方案：名额制 + 优先队列（推荐快速实现）

#### 6.3.1 适用场景

|场景特征|推荐方案|
|---|---|
|VM 数量 < 100|名额制（O(n) 完全够用）|
|快速原型验证|名额制（代码量 80 行，1 天完成）|
|需要公平性但非极致性能|名额制 + 优先队列（150 行，2-3 天）|
|跨平台兼容（Windows&Linux）|名额制 + std::set（放弃 OS 原生机制）|
|追求极致性能（生产环境）|完整 CFS+EDF+MLFQ（3000+ 行，3-5 周）|

#### 6.3.2 核心设计：名额制轮转

**基本思想**：
1. 维护 `std::unordered_map<uint64_t, VmSlot> vm_slots_` 标记每个 VM 的运行名额
2. 每个调度周期（如 10ms）重新分配名额
3. VM 线程检查自己有名额才运行，否则休眠等待
4. 时间片用完后主动让出，等待下一轮分配

``cpp
// VmManager.h - 名额制调度器（简化版）
class SimpleQuotaScheduler {
    struct VmSlot {
        std::shared_ptr<mVM> vm;
        bool has_quota = false;      // 当前是否有运行名额
        int priority = 0;            // 优先级（可选）
        uint64_t instructions_executed = 0;  // 统计信息
    };
    
    std::unordered_map<uint64_t, VmSlot> vm_slots_;  // VM 名额映射表
    std::atomic<bool> running_{false};
    int time_slice_ms_ = 10;         // 调度周期（毫秒）
    
public:
    void register_vm(uint64_t vm_id, std::shared_ptr<mVM> vm) {
        vm_slots_[vm_id] = {vm, false, 0, 0};
    }
    
    void start() {
        running_ = true;
        std::thread([this] { scheduler_loop(); }).detach();
    }
    
private:
    void scheduler_loop() {
        while (running_) {
            auto start_time = std::chrono::steady_clock::now();
            
            // 阶段 1：回收所有名额
            for (auto& [vm_id, slot] : vm_slots_) {
                slot.has_quota = false;
            }
            
            // 阶段 2：按策略分配名额（简单轮询）
            allocate_quotas();
            
            // 阶段 3：通知有名额的 VM 运行
            for (auto& [vm_id, slot] : vm_slots_) {
                if (slot.has_quota && slot.vm->isRunnable()) {
                    slot.vm->grant_quota();
                }
            }
            
            // 阶段 4：等待下一个调度周期
            auto elapsed = std::chrono::steady_clock::now() - start_time;
            auto sleep_time = std::max(
                std::chrono::milliseconds(0),
                time_slice_ms_ - elapsed
            );
            std::this_thread::sleep_for(sleep_time);
        }
    }
    
    void allocate_quotas() {
        // 简单轮询：每个可运行的 VM 都给名额
        // 可扩展为：按优先级、权重、老化机制等
        for (auto& [vm_id, slot] : vm_slots_) {
            if (slot.vm->get_state() == VMState::RUNNING ||
                slot.vm->get_state() == VMState::READY) {
                slot.has_quota = true;
            }
        }
    }
};

// mVM.h - 支持名额制的 VM 扩展
class mVM {
private:
    std::atomic<bool> quota_granted_{false};  // 是否被授予运行名额
    
public:
    void grant_quota() { quota_granted_ = true; }
    void revoke_quota() { quota_granted_ = false; }
    bool isRunnable() const { 
        return state_ == VMState::RUNNING || state_ == VMState::READY; 
    }
    
    void run_loop() {
        while (running_) {
            // 检查是否有运行名额
            if (!quota_granted_) {
                std::this_thread::sleep_for(std::chrono::milliseconds(1));
                continue;
            }
            
            // 有名额，执行指令（时间片内）
            for (int i = 0; i < 50 && running_; i++) {
                if (!execute_instruction()) break;
            }
            
            // 主动让出名额（不立即 revoke，由 scheduler 决定）
        }
    }
};
```

#### 6.3.3 性能优化：引入优先队列/PBDS

当 VM 数量 > 100 或需要更公平的调度时，可在名额制基础上增加优先队列：

**方案 A：使用 GCC PBDS（推荐，性能最优）**

``cpp
#include <ext/pb_ds/assoc_container.hpp>
#include <ext/pb_ds/tree_policy.hpp>
using namespace __gnu_pbds;

class PbdsQuotaScheduler {
    struct PriorityTask {
        uint64_t vm_id;
        int priority;
        int64_t vruntime;  // 虚拟运行时间（CFS 风格）
        
        bool operator<(const PriorityTask& other) const {
            if (priority != other.priority) {
                return priority > other.priority;  // 优先级高者优先
            }
            return vruntime < other.vruntime;      // 虚拟时间少者优先
        }
    };
    
    // PBDS 红黑树（自动平衡，O(log n) 操作）
    typedef tree<PriorityTask, null_type, std::less<PriorityTask>,
                 rb_tree_tag, tree_order_statistics_node_update>
        priority_tree_t;
    
    priority_tree_t ready_queue;
    std::unordered_map<uint64_t, PriorityTask> vm_priority_map;
    std::mutex mtx;
    
public:
    void enqueue_vm(uint64_t vm_id, int priority, int64_t vruntime) {
        std::lock_guard<std::mutex> lock(mtx);
        
        if (vm_priority_map.count(vm_id)) {
            ready_queue.erase(vm_priority_map[vm_id]);
        }
        
        PriorityTask task{vm_id, priority, vruntime};
        ready_queue.insert(task);
        vm_priority_map[vm_id] = task;
    }
    
    void grant_quotas_to_top_n(int n) {
        std::lock_guard<std::mutex> lock(mtx);
        
        int count = 0;
        for (auto it = ready_queue.begin(); 
             it != ready_queue.end() && count < n; 
             ++it, ++count) {
            vm_slots_[it->vm_id].has_quota = true;
        }
    }
    
    // PBDS 特有功能：查询排名
    int get_vm_rank(uint64_t vm_id) {
        std::lock_guard<std::mutex> lock(mtx);
        if (!vm_priority_map.count(vm_id)) return -1;
        return ready_queue.order_of_key(vm_priority_map[vm_id]);
    }
};
```

**方案 B：使用 std::set（跨平台兼容，MSVC 可用）**

``cpp
class StdSetQuotaScheduler {
    struct Task {
        uint64_t vm_id;
        int64_t key;  // deadline 或 vruntime
        bool operator<(const Task& other) const {
            return key < other.key;
        }
    };
    
    std::set<Task> ready_set;
    std::unordered_map<uint64_t, std::set<Task>::iterator> vm_index;
    std::mutex mtx;
    
public:
    void enqueue(uint64_t vm_id, int64_t key) {
        std::lock_guard<std::mutex> lock(mtx);
        
        if (vm_index.count(vm_id)) {
            ready_set.erase(vm_index[vm_id]);
        }
        
        auto result = ready_set.insert({vm_id, key});
        vm_index[vm_id] = result.first;
    }
    
    uint64_t select_next() {
        std::lock_guard<std::mutex> lock(mtx);
        if (ready_set.empty()) return INVALID_VM_ID;
        
        auto it = ready_set.begin();  // O(1) 最小元素
        uint64_t vm_id = it->vm_id;
        
        vm_index.erase(vm_id);
        ready_set.erase(it);
        
        return vm_id;
    }
};
