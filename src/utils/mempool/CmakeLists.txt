cmake_minimum_required(VERSION 3.10)
project(MemoryPool)

# C++17 标准
set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

# 包含头文件目录
include_directories(${PROJECT_SOURCE_DIR}/include)

# 源文件
set(SOURCES
    main.cpp
    src/RBtree.cpp
)

# 创建可执行文件
add_executable(mempool_demo ${SOURCES})

# Windows 特定设置
if(WIN32)
    target_compile_definitions(mempool_demo PRIVATE _CRT_SECURE_NO_WARNINGS)
endif()
